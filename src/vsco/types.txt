export interface FeatureFlag<T = unknown> {
	defaultValue: T;
	rules?: FeatureRule<T>[];
}

export interface FeatureRule<T = unknown> {
	condition?: Condition;
	parentConditions?: ParentCondition[];
	force?: T;

	coverage?: number;
	hashAttribute?: string;
	hashVersion?: number;
	bucketVersion?: number;
	seed?: string;
	gate?: boolean;
}

export interface ParentCondition {
	id: string;
	condition: Condition;
	gate?: boolean;
}

export type Condition =
	| PrimitiveCondition
	| LogicalCondition
	| ComparisonConditionMap;

export interface PrimitiveCondition {
	[key: string]: Primitive;
}

export interface LogicalCondition {
	$or?: Condition[];
	$and?: Condition[];
	$not?: Condition;
}

export interface ComparisonConditionMap {
	[key: string]:
		| Primitive
		| ComparisonOperators
		| Condition[]
		| undefined;
}

export interface ComparisonOperators {
	$in?: Primitive[];
	$nin?: Primitive[];

	$lt?: Primitive;
	$lte?: Primitive;
	$gt?: Primitive;
	$gte?: Primitive;

	$vlt?: string;
	$vlte?: string;
	$vgt?: string;
	$vgte?: string;

	$regex?: string;

	$exists?: boolean;
}

export type Primitive =
	| string
	| number
	| boolean
	| null;
